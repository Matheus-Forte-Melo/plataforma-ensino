    int main() {
        int A = 60, B = 6, C = 4;
    
        A = (A * B) + (C + 2 * 4); 
        B = (B - C) + A / 2;        
        A = A + (C + C) - 300;      
        B = B + (A - C);            
        C = C * C + 10;             
    
        // Resultado final:
        printf("A = %d, B = %d, C = %d\n", A, B, C);
        return 0;
    }
