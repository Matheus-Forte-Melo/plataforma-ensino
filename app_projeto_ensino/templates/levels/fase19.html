#include <stdio.h>
int main() {
    int a = 7, b = 3, c = 10;

    a = a + b * 2;
    b = a - c;
    c = a % 5;

    if (a > 10 && b < 5) {
        b = b * 2;
        a = a - 3;
    } else {
        c = c + 5;
    }

    if (b == 6) {
        a = a / 2;
        c = c * 3;
    } else {
        a = a + c;
    }

    if (c > 5 || b < 4) {
        c = c - 4;
        b = b + 2;
    } else {
        a = a + 1;
    }

    printf("Valores finais: a = %d, b = %d, c = %d\n", a, b, c);
    return 0;
}
